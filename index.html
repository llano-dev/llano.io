API PREVIEW

Simple to Call. Rich in Detail.

One endpoint. Bring your own route or let us plan it.

route_score.py 
import llano

client = llano.Client(api_key="sk_live_••••••••")

# Option A — pass a geometry from Samsara, TomTom, etc.
result = client.routes.score(
    geometry      = route_geojson,
    vehicle_class = "class_8",
    depart_at     = "2026-03-15T06:00"
)

# Option B — just give us O&D, we'll route it
result = client.routes.score(
    origin        = "4710 N PanAm Expy, San Antonio, TX",
    destination   = "250 University, Round Rock, TX",
    vehicle_class = "class_8",
    depart_at     = "2026-03-15T06:00"
)

# Same response either way
print(result.score)       # 41.2
print(result.grade)       # "FAIR"
print(result.routing)     # "provided" | "llano"

for h in result.hotspots:
    print(h.score, h.reason)
# 28  "construction"
# 35  "weather"
import llano

client = llano.Client(api_key="sk_live_••••••••")

# Mix geometries and O&D in the same batch
results = client.routes.batch(
    routes = [
        {"geometry": route_a_geojson},
        {"geometry": route_b_geojson},
        {"origin": "Houston, TX", "destination": "Dallas, TX"},
    ],
    vehicle_class = "class_8",
    depart_at     = "2026-03-15T06:00"
)

# Ranked by safety score, highest first
for r in results.ranked():
    print(r.score, r.grade, r.routing)
